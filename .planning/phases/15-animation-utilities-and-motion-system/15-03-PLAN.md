**useShimmer.ts** — Shared shimmer animation driver (UI-SPEC Section 3.8, D-14):

```typescript
import { useEffect } from 'react';
import {
  useSharedValue,
  withRepeat,
  withTiming,
  cancelAnimation,
  Easing,
  useReducedMotion,
} from 'react-native-reanimated';
import { TIMING_SHIMMER } from '@/src/constants/springs';

export function useShimmer(width: number) {
  const translateX = useSharedValue(-width);
  const reducedMotion = useReducedMotion();

  useEffect(() => {
    if (reducedMotion) return;
    translateX.value = -width;
    translateX.value = withRepeat(
      withTiming(width, TIMING_SHIMMER),
      -1,    // infinite
      false  // no reverse
    );
    return () => { cancelAnimation(translateX); };
  }, [width, reducedMotion]);

  return translateX;
}
```

Key requirements:
- Accepts `width` parameter to know sweep distance
- Returns `translateX` shared value for consumers
- `withRepeat(-1, false)` for infinite left-to-right sweep (D-14)
- `cancelAnimation(translateX)` in useEffect cleanup to prevent memory leak (Pitfall 4 from RESEARCH)
- When reduced motion is on, skip animation entirely (static placeholder)
- Import TIMING_SHIMMER (duration: 1200, easing: linear) from springs.ts
